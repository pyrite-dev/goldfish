635int gf_lua_run(gf_lua_t* lua, const char* path) {
636 gf_file_t* f = gf_file_open(lua->engine, path, "r");
637 char* str;
638 if(f == NULL) return -1;
639 str = malloc(f->size + 1);
640 gf_file_read(f, str, f->size);
641 str[f->size] = 0;
642
643 if(luaL_loadstring(lua->lua, str)) {
644 gf_log_function(lua->engine, "Lua error: %s", lua_tostring(lua->lua, -1));
645 free(str);
646 gf_file_close(f);
647 return -1;
648 }
649
650 if(lua_pcall(lua->lua, 0, LUA_MULTRET, 0)) {
651 gf_log_function(lua->engine, "Lua error: %s", lua_tostring(lua->lua, -1));
652 free(str);
653 gf_file_close(f);
654 return -1;
655 }
656
657 free(str);
658 gf_file_close(f);
659 return 0;
660}
